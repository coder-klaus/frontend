```js
import express from 'express'

// 创建一个express应用
// 底层基于 http模块的 createServer 方法
const app = express()

// express默认支持 后端路径 => 即根据请求方法和请求路径来匹配对应controller进行处理
app.get('/', (req, res) => {
  // req 和 res 是基于原生 http模块的 req和res的二次封装对象
  res.send('Hello World')
})

// 启动一个http服务，监听3000端口
app.listen(3000, () => {
  // 当服务启动后，会执行回调函数
  console.log('Server is running on port 3000')
})
```
