## end

```js
// 使用原生request对象的end方法返回响应体
app.get('/', (req, res) => {
  // 需要自己根据返回的内容手动设置合适的Content-Type => 否则默认会被解析为 text/plain
  // 需要手动声明返回内容对应编码格式为 utf-8，否则不同客户端使用的编码格式可能不同，从而导致解析结果不一致
  // + 浏览器环境，将使用 GBK 进行解码，从而导致乱码 「 你好 => 浣犲ソ 」
  // + postmane 将使用 utf-8 进行解码，从而导致乱码 「 你好 => 你好 」
  res.setHeader('Content-Type', 'text/html; charset=utf-8')
  res.end('你好')
})
```



## json

```js
import express from 'express'

const app = express()

app.get('/', (req, res) => {
  // 返回 json 对象
  // 自动设置 Content-Type 为 application/json; charset=utf-8
  res.json({
    code: 200,
    message: '你好'
  })
})

app.listen(3000)
```



## send

> 虽然 `send`也可以返回 `json`格式数据，但依旧推荐返回`json`格式数据时，使用`res.json`
>
> 这样语义化更强一些

```js
import express from 'express'

const app = express()

app.get('/', (req, res) => {
  // send 方法会根据返回内容自动设置 Content-Type
  // 1. 如果返回内容为字符串，则 Content-Type 为 text/html; charset=utf-8
  // 2. 如果返回内容为二进制值，则 Content-Type 为 application/octet-stream
  // 3. 其余情况下，则 Content-Type 为 application/json; charset=utf-8
  res.send({
    code: 200,
    message: '你好'
  })
})

app.listen(3000)
```



## status

```js
import express from 'express'

const app = express()

app.get('/', (req, res) => {
  // 可以通过 status 方法设置响应状态码
  // status 方法本身返回 res 对象，因此可以链式调用
  res.status(201).send('创建成功~')
})

app.listen(3000)
```
