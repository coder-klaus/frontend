ESM默认只能存在一个默认导出，默认导出不携带名称，导入时可以自己任意命名

```js
// 具名导出 => 可以有多个
export const add = (a, b) => a + b
export const sub = (a, b) => a - b

// 默认导出 => 只能有一个
export default (a, b) => a * b
```

```js
// 具名导入 和 默认导入 可以混合使用
// 1. mul 是默认导出的，所以可以随意命名
// 2. add 和 sub 是具名导出的，所以需要按照具名导出的方式来导入
import mul, { add, sub } from './utils/math.js'

console.log(add(1, 2))
console.log(sub(1, 2))
console.log(mul(1, 2))
```



```js
import * as math from './utils/math.js'

console.log(math)
/*
{
  add: [Function: add],
  sub: [Function: sub],
  // * as math时，会将导出内容转换为对象
  // 其中，默认导出会转换为 default 属性
  default: [Function: mul]
}
*/
```
