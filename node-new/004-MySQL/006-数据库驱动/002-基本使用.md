## 基本使用

```js
import mysql from 'mysql2';

// 初始化数据库连接
// 通常情况下，host 应为远程数据库服务器的 IP 地址
// user 参数应为拥有特定访问权限的数据库子账户
// 出于安全原因，root 账户一般禁止通过远程 IP 登录数据库
const connection = mysql.createConnection({
  host: 'localhost',
  user: 'root',
  password: 'P@ssw0rd',
  database: 'demo_db'
})

// 编写SQL语句 「 对应变量一般命名为 statement 居多 」
const statement = 'SELECT * FROM `students`';

// 通过链接执行SQL语句
// 参数一 => 需要执行的SQL语句
// 参数二 => 回调函数
connection.query(statement, (err, results, fields) => {
  // 回调函数接收三个参数
  // 1. 参数一 => 错误信息 => 如果没有错误，则返回 null
  // 2. 参数二 => 查询结果 => 一般是对象数组
  // 3. 参数三 => 字段信息 => 字段信息构成的数组 「 较少用 」
  if (err) throw err;
  console.log('results', results);
  /*
   results [
      { student_id: 1, name: 'Alice' },
      { student_id: 2, name: 'Bob' },
      { student_id: 3, name: 'Jhon' }
   ]
  */
  console.log('fields', fields);
  /*
    fields [
      `student_id` INT NOT NULL PRIMARY KEY,
      `name` VARCHAR(50)
    ]
  */
})

// 关闭连接
connection.end();
```
