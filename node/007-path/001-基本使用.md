用于解析路径的 node 内置模块

```js
import path from 'node:path'
import { fileURLToPath } from 'node:url'

// ESM 中 每一个模块都是一个 URL
console.log(import.meta.url) // => file:///Users/klaus/Desktop/node-demo/foo.js

// fileURLToPath => 将 ESM 中的 URL 转换为 普通文件路径
// + CJS 中存在全局变量 __filename 和 __dirname，
// + 在 ESM 中 需要使用 fileURLToPath 和 path.dirname 来获取
const __filename = fileURLToPath(import.meta.url) // => /Users/klaus/Desktop/node-demo/foo.js
const __dirname = path.dirname(__filename) // => /Users/klaus/Desktop/node-demo
```

```js
import path from 'node:path'
import { fileURLToPath } from 'node:url'

const __filename = fileURLToPath(import.meta.url)  // => /Users/klaus/Desktop/node-demo/foo.js
const __dirname = path.dirname(__filename) // => /Users/klaus/Desktop/node-demo

// 获取文件名
console.log(path.basename(__filename)) // foo.js
// 可以传入第二个参数，会自动从最终结果中移除
console.log(path.basename(__filename, '.js')) // foo

// 获取文件路径 =>
console.log(path.dirname(__filename)) // /Users/klaus/Desktop/node-demo

// 获取文件扩展名 => 带 . 前缀
console.log(path.extname(__filename)) // .js

// 将路径解析为文件对象
const fileObj = path.parse(__filename)
console.log(fileObj)
/*
  {
    root: '/',
    dir: '/Users/klaus/Desktop/node-demo',
    base: 'foo.js',
    ext: '.js',
    name: 'foo'
  }
*/

// 将文件对象转为路径字符串
console.log(path.format(fileObj)) // /Users/klaus/Desktop/node-demo/foo.js

// 将路径转为绝对路径
console.log(path.resolve('./src', 'bar')) // /Users/klaus/Desktop/node-demo/bar

// 判断路径是否为绝对路径
console.log(path.isAbsolute(__filename)) // true
```
