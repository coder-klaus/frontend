```js
app.use((req, res) => {
  // res本质就是响应流，所以可以通过 res.write + res.end 来响应结果
  // 默认响应头的 Content-Type 为 text/plain
  res.end('Hello World')
})
```

```js
app.use((req, res) => {
  // 1. 返回结果为JSON格式字符串
  // 2. 参数是JSON对象，会自动序列化
  // 3. 设置响应头 Content-Type: application/json
  res.json({
    name: 'Klaus',
    age: 23
  })
})
```

```js
app.use((req, res) => {
  // send方法会根据参数自动设置 Content-Type
  // 1. 字符串 => text/html
  // 2. 二进制 「 Buffer 」 => application/octet-stream
  //    + octet 表示的是 网络传输中任意的8位字节流，即一个字节
  // 3. 其余数据类型 => application/json
  const buf = Buffer.from('Hello World')
  res.send(buf)
})
```

```js
import express from 'express'

const app = new express()

app.use((req, res) => {
  // 可以通过 status方法设置响应码，只是设置响应码，设置完成后还需手动关闭响应留
  // 1. 如果响应码设置正确，会自动识别响应码描述
  // 2. 如果响应码设置错误，响应码描述即为 UNKNOWN
  res.status(333).send('end')
})

app.listen(3000)
```
